function g4(){ let r={x:200,y:500,vy:0}; let fuel=100; let obs=[],met=[]; let dist=0; // límites físicos (IMPORTANTE) const gravity = 0.03; const thrust = 0.18; const maxSpeed = 6; const drag = 0.99; function loop(){ ctx.fillStyle="#000"; ctx.fillRect(0,0,c.width,c.height); dist++; /* spawn enemigos */ if(Math.random()<0.06){ obs.push({x:Math.random()*c.width,y:-50}); } if(Math.random()<0.08){ met.push({x:Math.random()*c.width,y:-50,vy:2+Math.random()*3}); } /* CONTROLES MEJORADOS */ let pushing = false; if(keys.w && fuel>0){ r.vy -= thrust; fuel -= 0.15; pushing = true; } if(keys.a) r.x -= 4; if(keys.d) r.x += 4; /* FÍSICA CORREGIDA */ r.vy += gravity; /* freno natural (clave del bug arreglado) */ r.vy *= drag; /* límite de velocidad */ if(r.vy > maxSpeed) r.vy = maxSpeed; if(r.vy < -maxSpeed) r.vy = -maxSpeed; /* si no empujas, estabiliza más rápido */ if(!pushing && Math.abs(r.vy) < 0.2){ r.vy = 0; } /* movimiento */ r.y += r.vy; /* COHETE */ ctx.fillStyle="orange"; ctx.fillRect(r.x,r.y,20,40); /* OBSTÁCULOS */ ctx.fillStyle="red"; for(let i=0;i obs[i].x && r.y < obs[i].y+40 && r.y+40 > obs[i].y ){ over("💥 Choque con obstáculo"); return; } } /* METEORITOS */ ctx.fillStyle="white"; for(let i=0;i met[i].x && r.y < met[i].y+10 && r.y+40 > met[i].y ){ over("☄️ Impacto de meteorito"); return; } } /* HUD */ ctx.fillStyle="white"; ctx.fillText("Fuel: "+fuel.toFixed(0),10,20); ctx.fillText("Distancia: "+dist,10,40); /* VICTORIA */ if(dist>5000){ over("🌙 ¡Llegaste a la Luna!"); return; } requestAnimationFrame(loop); } loop(); }
